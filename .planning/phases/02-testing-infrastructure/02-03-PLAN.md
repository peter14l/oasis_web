import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import RazorpayPayment from './RazorpayPayment';

// Mock Supabase
vi.mock('../supabaseClient', () => ({
  supabase: {
    auth: {
      getUser: vi.fn().mockResolvedValue({
        data: { user: { email: 'test@test.com', user_metadata: { full_name: 'Test' } } }
      }),
    },
    functions: {
      invoke: vi.fn(),
    },
  },
}));

import { supabase } from '../supabaseClient';

// Mock window.Razorpay
const mockRazorpay = {
  open: vi.fn(),
};
window.Razorpay = vi.fn().mockImplementation(() => mockRazorpay);

describe('RazorpayPayment', () => {
  beforeEach(() => {
    vi.clearAllMocks();
    // Reset script loaded state by re-rendering
  });

  afterEach(() => {
    // Clean up script
    const script = document.getElementById('razorpay-checkout');
    if (script) script.remove();
  });

  it('renders button with Razorpay text', () => {
    render(<RazorpayPayment plan="pro" amount={100} currency="INR" onSuccess={() => {}} />);
    expect(screen.getByText(/razorpay/i)).toBeInTheDocument();
  });

  it('loads Razorpay script on mount', async () => {
    render(<RazorpayPayment plan="pro" amount={100} currency="INR" onSuccess={() => {}} />);
    
    await waitFor(() => {
      const script = document.querySelector('script[src*="razorpay"]');
      expect(script).toBeInTheDocument();
    });
  });

  it('calls create-order when payment initiated', async () => {
    const mockOrderData = { id: 'order_123', amount: 10000, currency: 'INR' };
    supabase.functions.invoke.mockResolvedValueOnce({ data: mockOrderData, error: null });
    
    render(<RazorpayPayment plan="pro" amount={100} currency="INR" onSuccess={() => {}} />);
    
    // Wait for script to load
    await waitFor(() => {
      expect(document.querySelector('script[src*="razorpay"]')).toBeInTheDocument();
    });
    
    // Click payment button
    await userEvent.click(screen.getByRole('button', { name: /razorpay/i }));
    
    expect(supabase.functions.invoke).toHaveBeenCalledWith('razorpay-create-order', {
      body: { plan: 'pro', amount: 100, currency: 'INR' }
    });
  });

  it('handles payment error', async () => {
    const mockOrderData = { id: 'order_123', amount: 10000, currency: 'INR' };
    supabase.functions.invoke.mockResolvedValueOnce({ data: mockOrderData, error: null });
    
    const alertSpy = vi.spyOn(window, 'alert').mockImplementation(() => {});
    
    render(<RazorpayPayment plan="pro" amount={100} currency="INR" onSuccess={() => {}} />);
    
    await waitFor(() => {
      expect(document.querySelector('script[src*="razorpay"]')).toBeInTheDocument();
    });
    
    await userEvent.click(screen.getByRole('button', { name: /razorpay/i }));
    
    // Simulate Razorpay handler error
    const rzp = mockRazorpay.open.mock.calls[0][0];
    rzp.handler({ razorpay_order_id: 'order_123', razorpay_payment_id: 'pay_123', razorpay_signature: 'invalid' });
    
    expect(alertSpy).toHaveBeenCalledWith('Payment verification failed.');
    
    alertSpy.mockRestore();
  });

  it('calls onSuccess on payment verified', async () => {
    const mockOrderData = { id: 'order_123', amount: 10000, currency: 'INR' };
    supabase.functions.invoke
      .mockResolvedValueOnce({ data: mockOrderData, error: null })
      .mockResolvedValueOnce({ error: null }); // verify call
    
    const onSuccess = vi.fn();
    
    render(<RazorpayPayment plan="pro" amount={100} currency="INR" onSuccess={onSuccess} />);
    
    await waitFor(() => {
      expect(document.querySelector('script[src*="razorpay"]')).toBeInTheDocument();
    });
    
    await userEvent.click(screen.getByRole('button', { name: /razorpay/i }));
    
    const rzp = mockRazorpay.open.mock.calls[0][0];
    rzp.handler({ razorpay_order_id: 'order_123', razorpay_payment_id: 'pay_123', razorpay_signature: 'valid' });
    
    await waitFor(() => {
      expect(onSuccess).toHaveBeenCalled();
    });
  });
});

import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { render, screen, waitFor, act } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import PayPalPayment from './PayPalPayment';

// Mock Supabase
vi.mock('../supabaseClient', () => ({
  supabase: {
    functions: {
      invoke: vi.fn(),
    },
  },
}));

import { supabase } from '../supabaseClient';

// Mock window.paypal
const mockPaypal = {
  Buttons: vi.fn().mockReturnValue({
    render: vi.fn().mockResolvedValue(undefined),
  }),
};
window.paypal = mockPaypal;

describe('PayPalPayment', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  afterEach(() => {
    // Clean up script
    const script = document.getElementById('paypal-sdk-INR');
    if (script) script.remove();
  });

  it('shows loading state initially', () => {
    render(<PayPalPayment plan="pro" amount={10} currency="USD" onSuccess={() => {}} />);
    expect(screen.getByText(/loading/i)).toBeInTheDocument();
  });

  it('loads PayPal SDK script', async () => {
    render(<PayPalPayment plan="pro" amount={10} currency="USD" onSuccess={() => {}} />);
    
    await waitFor(() => {
      const script = document.getElementById('paypal-sdk-INR');
      expect(script).toBeInTheDocument();
    });
  });

  it('calls createOrder with correct amount', async () => {
    render(<PayPalPayment plan="pro" amount={10} currency="USD" onSuccess={() => {}} />);
    
    await waitFor(() => {
      expect(mockPaypal.Buttons).toHaveBeenCalled();
    });
    
    const buttonsConfig = mockPaypal.Buttons.mock.calls[0][0];
    const order = await buttonsConfig.createOrder();
    
    expect(order.purchase_units[0].amount.value).toBe('10.00');
    expect(order.purchase_units[0].description).toContain('pro');
  });

  it('handles onApprove and verifies payment', async () => {
    const onSuccess = vi.fn();
    supabase.functions.invoke.mockResolvedValueOnce({ error: null });
    
    render(<PayPalPayment plan="pro" amount={10} currency="USD" onSuccess={onSuccess} />);
    
    await waitFor(() => {
      expect(mockPaypal.Buttons).toHaveBeenCalled();
    });
    
    const buttonsConfig = mockPaypal.Buttons.mock.calls[0][0];
    
    await act(async () => {
      await buttonsConfig.onApprove({ orderID: 'order_123' }, {
        order: { purchase_units: [{ amount: { value: '10.00' } }] }
      });
    });
    
    expect(supabase.functions.invoke).toHaveBeenCalledWith('paypal-verify', {
      body: expect.objectContaining({ orderId: 'order_123', plan: 'pro' })
    });
    expect(onSuccess).toHaveBeenCalled();
  });

  it('handles verification error', async () => {
    supabase.functions.invoke.mockResolvedValueOnce({ error: { message: 'Verification failed' } });
    
    render(<PayPalPayment plan="pro" amount={10} currency="USD" onSuccess={() => {}} />);
    
    await waitFor(() => {
      expect(mockPaypal.Buttons).toHaveBeenCalled();
    });
    
    const buttonsConfig = mockPaypal.Buttons.mock.calls[0][0];
    
    await act(async () => {
      await buttonsConfig.onApprove({ orderID: 'order_123' }, {
        order: { purchase_units: [{ amount: { value: '10.00' } }] }
      });
    });
    
    await waitFor(() => {
      expect(screen.getByText(/verification failed/i)).toBeInTheDocument();
    });
  });
});

import { describe, it, expect, vi, beforeEach } from 'vitest';
import { render, screen, waitFor, act } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import PayUPayment from './PayUPayment';

// Mock Supabase
vi.mock('../supabaseClient', () => ({
  supabase: {
    auth: {
      getUser: vi.fn(),
    },
    functions: {
      invoke: vi.fn(),
    },
  },
}));

import { supabase } from '../supabaseClient';

describe('PayUPayment', () => {
  beforeEach(() => {
    vi.clearAllMocks();
    // Clean up any created forms
    document.body.querySelectorAll('form').forEach(f => f.remove());
  });

  it('renders button with plan name', () => {
    render(<PayUPayment plan="pro" amount={10} currency="INR" />);
    expect(screen.getByText(/pro/i)).toBeInTheDocument();
  });

  it('shows alert when user not authenticated', async () => {
    supabase.auth.getUser.mockResolvedValueOnce({ data: { user: null } });
    const alertSpy = vi.spyOn(window, 'alert').mockImplementation(() => {});
    
    render(<PayUPayment plan="pro" amount={10} currency="INR" />);
    
    await userEvent.click(screen.getByRole('button', { name: /pro/i }));
    
    expect(alertSpy).toHaveBeenCalledWith('Please login to subscribe.');
    alertSpy.mockRestore();
  });

  it('calls payu-init when authenticated', async () => {
    supabase.auth.getUser.mockResolvedValueOnce({
      data: { user: { email: 'test@test.com', user_metadata: { full_name: 'Test' } } }
    });
    supabase.functions.invoke.mockResolvedValueOnce({
      data: {
        payu_url: 'https://test.payu.in/_payment',
        key: 'test_key',
        txnid: 'txn_123',
        amount: '10',
        productinfo: 'pro',
        firstname: 'Test',
        email: 'test@test.com',
        phone: '0000000000',
        hash: 'hash123',
        surl: 'https://example.com/success',
        furl: 'https://example.com/failure',
      },
      error: null
    });
    
    render(<PayUPayment plan="pro" amount={10} currency="INR" />);
    
    await userEvent.click(screen.getByRole('button', { name: /pro/i }));
    
    await waitFor(() => {
      expect(supabase.functions.invoke).toHaveBeenCalledWith('payu-init', {
        body: expect.objectContaining({
          plan: 'pro',
          amount: 10,
          currency: 'INR',
        })
      });
    });
  });

  it('creates and submits form to PayU', async () => {
    supabase.auth.getUser.mockResolvedValueOnce({
      data: { user: { email: 'test@test.com', user_metadata: { full_name: 'Test' } } }
    });
    supabase.functions.invoke.mockResolvedValueOnce({
      data: {
        payu_url: 'https://test.payu.in/_payment',
        key: 'test_key',
        txnid: 'txn_123',
        amount: '10',
        productinfo: 'pro',
        firstname: 'Test',
        email: 'test@test.com',
        phone: '0000000000',
        hash: 'hash123',
        surl: 'https://example.com/success',
        furl: 'https://example.com/failure',
      },
      error: null
    });
    
    const submitSpy = vi.fn();
    vi.spyOn(HTMLFormElement.prototype, 'submit', 'get').mockImplementation(() => submitSpy());
    
    render(<PayUPayment plan="pro" amount={10} currency="INR" />);
    
    await userEvent.click(screen.getByRole('button', { name: /pro/i }));
    
    await waitFor(() => {
      const form = document.body.querySelector('form[action*="payu"]');
      expect(form).toBeInTheDocument();
      expect(submitSpy).toHaveBeenCalled();
    });
  });

  it('shows processing state when loading', async () => {
    supabase.auth.getUser.mockImplementationOnce(
      () => new Promise(resolve => setTimeout(() => resolve({ data: { user: null } }), 100))
    );
    
    render(<PayUPayment plan="pro" amount={10} currency="INR" />);
    
    await userEvent.click(screen.getByRole('button', { name: /pro/i }));
    
    expect(screen.getByText(/processing/i)).toBeInTheDocument();
  });
});
